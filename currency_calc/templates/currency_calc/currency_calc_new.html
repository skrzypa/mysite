import bs4
import json
import requests

def all_records() -> list[dict]:
    url = "https://www.pskrzynski.pl/currency_calc_records/"

    get = requests.get(url= url).text

    bs = bs4.BeautifulSoup(get, "html.parser") \
        .find_all('div', {'class': 'record'})

    records = []
    for b in bs:
        b: bs4.element.Tag

        s = str(b.contents[0]) \
            .strip() \
            .replace("'", '"')
        
        records.append(
            json.loads(s)
        )
    return records
